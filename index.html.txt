  <script>
    // Expose React + hooks as globals so the component's destructured import resolves after we strip it.
    window.React = React;
    ["useState","useEffect","useMemo","useRef","useCallback","useReducer","useContext","Fragment"]
      .forEach(function(n){ window[n] = React[n]; });
    window.__setApp = function(C){ window.__App = C; };
  </script>

  <script>
    fetch("App.jsx")
      .then(function(r){ if(!r.ok) throw new Error("App.jsx not found (HTTP "+r.status+")"); return r.text(); })
      .then(function(src){
        // 1) remove the ES module import line (hooks are provided as globals above)
        src = src.replace(/^\s*import[^\n]*\n/m, "");
        // 2) turn the default export into a plain function declaration
        src = src.replace(/export\s+default\s+function\s+App\s*\(/, "function App(");
        src = src.replace(/export\s+default\s+App\s*;?/, "");
        // 3) register the component
        src += "\n;window.__setApp(App);";
        // 4) compile JSX -> JS in the browser
        var out = Babel.transform(src, { presets: ["react"] }).code;
        var s = document.createElement("script");
        s.text = out;
        document.body.appendChild(s);
        if (window.__App) {
          ReactDOM.createRoot(document.getElementById("root")).render(React.createElement(window.__App));
        } else {
          document.getElementById("root").innerHTML = '<div class="boot">Could not start the app (App component not found).</div>';
        }
      })
      .catch(function(e){
        document.getElementById("root").innerHTML = '<div class="boot">Failed to load: ' + e.message + '</div>';
      });
  </script>
